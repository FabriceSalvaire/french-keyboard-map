\newcommand{\key}[6]{%
  \node (a) at ($ (#1*\width,#2*\height) + (#1*\corner,#2*\corner) $) {};
  \draw[white,fill=black,rounded corners=1mm] %
  ($ (a) $) -- ++(\width,0mm) -- ++(0mm,\height) -- ++(-\width,0mm) -- cycle;
  \draw[fill=black] %
  ($ (a) + (\width,\height) $) -- ++(\corner,0mm) -- ++(-\corner,\corner) -- cycle;
  \draw[white] ($ (a) + (1/3*\width,1/3*\height-.5mm) $) node {\myfont{\large{#3}}};
  \draw[white] ($ (a) + (1/3*\width,2/3*\height+.5mm) $) node {\myfont{\large{#4}}};
  \draw[white] ($ (a) + (2/3*\width,1/3*\height-.5mm) $) node {\myfont{\large{#5}}};
  \draw[white] ($ (a) + (2/3*\width,2/3*\height+.5mm) $) node {\myfont{\large{#6}}};
}

  ($ (a) $) -- ++(\width,0mm) -- ++(0mm,\height) -- ++(-\width,0mm) -- cycle;
  \draw[fill=black] %
  ($ (a) + (\width,\height) $) -- ++(\corner,0mm) -- ++(-\corner,\corner) -- cycle;
  \draw[white] ($ (a) + (1/3*\width,1/2*\height) $) node {\myfont{\large{#3}}};
  \draw[white] ($ (a) + (2/3*\width,1/2*\height) $) node {\myfont{\large{#4}}};
}
